start numSimilar(Strand s, Strand x){
	int similarity = 0;
	int totalSim = 0;
	int numSimCodons = 0;
	for (int i=0; i<3; i++){
		if (s[i] == x[i]){
			similarity++; //max of 3
			totalSim++;
		}
		if (similarity == 3)
			numSimCodons++;
	}
}
